k1=1.0
k2=1.0
tmax=100
def run(dt):
    traj=[]
    q=np.array((0,1,2))
    qold=np.array((0,1,2))
    f=np.zeros(3)
    for i in range(int(tmax/dt)):
        f[0]=k1*(q[1]-q[0])
        f[1]=k1*(q[0]-q[1])+k2*(q[2]-q[1])
        f[2]=k2*(q[1]-q[2])
        qnew=2*q-qold+f/m*dt**2
        qold=+q
        q=+qnew
        #print(f)
        traj.append((i*dt,q[0],q[1],q[2]))
    traj=np.array(traj)
    return traj
